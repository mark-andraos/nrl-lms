// ============================================================ // processMatchResults — call this inside updateNRLResults // after a match is marked as Final // ============================================================ function processMatchResults() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const matchesSheet = ss.getSheetByName("Matches"); const tipsSheet = ss.getSheetByName("Tips"); const entrantsSheet = ss.getSheetByName("Entrants"); const resultsSheet = ss.getSheetByName("Results"); const matchesData = matchesSheet.getDataRange().getValues(); const tipsData = tipsSheet.getDataRange().getValues(); const entrantsData = entrantsSheet.getDataRange().getValues(); const resultsData = resultsSheet.getDataRange().getValues(); // Build a set of already-processed result keys (EntrantID-Round) const processedKeys = new Set(); for (let i = 4; i < resultsData.length; i++) { if (resultsData[i][8]) processedKeys.add(String(resultsData[i][8])); } // Loop through all finished matches for (let i = 4; i < matchesData.length; i++) { const row = matchesData[i]; const finished = row[11]; const isFinished = finished === true || finished === 'TRUE' || finished === 'true' || finished === 1; if (!isFinished) continue; const round = Number(row[1]); const winner = String(row[8]).trim(); const loser = String(row[9]).trim(); if (!loser) continue; // skip draws for now — handle separately if needed // Handle draws — loser field may contain "Team1,Team2" const losers = loser.split(',').map(l => l.trim()).filter(Boolean); // Find all tips for this round for (let j = 4; j < tipsData.length; j++) { const tip = tipsData[j]; if (!tip[0]) continue; const entrantID = String(tip[2]); const tipRound = Number(tip[4]); const tipTeam = String(tip[5]).trim(); if (tipRound !== round) continue; const resultKey = entrantID + '-' + round; if (processedKeys.has(resultKey)) continue; // already processed // Determine if this tip won or lost const tipWon = tipTeam === winner; const tipLost = losers.includes(tipTeam); if (!tipWon && !tipLost) continue; // match not relevant to this tip const eliminated = tipLost; // Write to Results sheet const resultID = Utilities.getUuid().substring(0, 8); resultsSheet.appendRow([ resultID, 'COMP001', entrantID, tip[3], // email round, tipTeam, eliminated, new Date(), resultKey ]); processedKeys.add(resultKey); Logger.log((eliminated ? '❌ Eliminated' : '✅ Survived') + ': ' + entrantID + ' (' + tipTeam + ') Round ' + round); // If eliminated, update Alive = FALSE in Entrants sheet if (eliminated) { for (let k = 4; k < entrantsData.length; k++) { if (String(entrantsData[k][0]) === entrantID) { entrantsSheet.getRange(k + 1, 5).setValue(false); entrantsData[k][4] = false; // update local cache too Logger.log('Marked as eliminated in Entrants: ' + entrantsData[k][2]); break; } } } } } Logger.log('processMatchResults complete.'); }
